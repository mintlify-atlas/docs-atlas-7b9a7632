# VTextField

The `VTextField` component provides a single-line text input field with support for various input types, validation, prefix/suffix text, and more.

## Basic Usage

```vue
<template>
  <VTextField
    v-model="name"
    label="Name"
  />
</template>

<script setup>
import { ref } from 'vue'

const name = ref('')
</script>
```

## Props

<ParamField path="modelValue" type="string | number">
  The value of the text field.
</ParamField>

<ParamField path="label" type="string">
  Label text for the text field.
</ParamField>

<ParamField path="placeholder" type="string">
  Placeholder text.
</ParamField>

<ParamField path="type" type="string" default="text">
  HTML input type (text, password, email, number, tel, url, search, etc.).
</ParamField>

<ParamField path="prefix" type="string">
  Text to prepend inside the input.
</ParamField>

<ParamField path="suffix" type="string">
  Text to append inside the input.
</ParamField>

<ParamField path="counter" type="boolean | number | string">
  Display a character counter. Pass a number to set max length.
</ParamField>

<ParamField path="counterValue" type="number | ((value: any) => number)">
  Function or number to compute the counter value.
</ParamField>

<ParamField path="persistentCounter" type="boolean" default="false">
  Always show the counter instead of only when focused.
</ParamField>

<ParamField path="persistentPlaceholder" type="boolean" default="false">
  Always show the placeholder.
</ParamField>

<ParamField path="clearable" type="boolean" default="false">
  Add a clear icon to remove the text.
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Disable the text field.
</ParamField>

<ParamField path="readonly" type="boolean" default="false">
  Make the text field readonly.
</ParamField>

<ParamField path="autofocus" type="boolean" default="false">
  Automatically focus the field when mounted.
</ParamField>

<ParamField path="role" type="string">
  ARIA role attribute.
</ParamField>

## Character Counter

```vue
<template>
  <div>
    <!-- Simple counter -->
    <VTextField
      v-model="username"
      label="Username"
      counter
    />
    
    <!-- Counter with max length -->
    <VTextField
      v-model="bio"
      label="Bio"
      :counter="100"
      :rules="[v => v.length <= 100 || 'Max 100 characters']"
    />
    
    <!-- Always visible counter -->
    <VTextField
      v-model="title"
      label="Title"
      counter
      persistent-counter
    />
  </div>
</template>

<script setup>
import { ref } from 'vue'

const username = ref('')

</script>
```

## Validation

```vue
<template>
  <VForm>
    <VTextField
      v-model="email"
      label="Email"
      type="email"
      :rules="[
        v => !!v || 'Email is required',
        v => /.+@.+\..+/.test(v) || 'Email must be valid'
      ]"
    />
    
    <VTextField
      v-model="username"
      label="Username"
      :counter="20"
      :rules="[
        v => !!v || 'Username is required',
        v => (v && v.length >= 3) || 'Min 3 characters',
        v => (v && v.length <= 20) || 'Max 20 characters',
        v => /^[a-zA-Z0-9_]+$/.test(v) || 'Only letters, numbers, and underscores'
      ]"
    />
    
    <VTextField
      v-model="age"
      label="Age"
      type="number"
      :rules="[
        v => !!v || 'Age is required',
        v => v > 0 || 'Age must be positive',
        v => v < 150 || 'Age must be realistic'
      ]"
    />
  </VForm>
</template>

<script setup>
import { ref } from 'vue'

const email = ref('')
const username = ref('')
const age = ref('')
</script>
```

## Password Toggle

```vue
<template>
  <VTextField
    v-model="password"
    label="Password"
    :type="showPassword ? 'text' : 'password'"
    :append-inner-icon="showPassword ? 'mdi-eye' : 'mdi-eye-off'"
    @click:append-inner="showPassword = !showPassword"
  />
</template>

<script setup>
import { ref } from 'vue'

const showPassword = ref(false)
</script>
```

## Events

<ParamField path="update:modelValue" type="(value: string) => void">

<ParamField path="update:focused" type="(focused: boolean) => void">
