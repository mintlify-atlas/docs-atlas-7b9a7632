# VRadio & VRadioGroup

The `VRadio` component provides a single radio button, while `VRadioGroup` manages a group of radio buttons for mutually exclusive selection.

## Basic Usage

```vue
<template>
  <VRadioGroup v-model="selected" label="Select an option">
    <VRadio label="Option 1" value="1" />
    <VRadio label="Option 2" value="2" />
    <VRadio label="Option 3" value="3" />
  </VRadioGroup>
</template>

<script setup>
import { ref } from 'vue'

const selected = ref('1')
</script>
```

## VRadioGroup Props

<ParamField path="modelValue" type="any">
  The selected value from the radio group.
</ParamField>

<ParamField path="label" type="string">
  Label for the radio group.
</ParamField>

<ParamField path="column" type="boolean" default="true">
  Display radios in a column layout.
</ParamField>

<ParamField path="row" type="boolean" default="false">
  Display radios in a row layout.
</ParamField>

<ParamField path="inline" type="boolean" default="false">
  Display radios inline (wrapping).
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Disable all radio buttons in the group.
</ParamField>

<ParamField path="readonly" type="boolean" default="false">
  Make all radio buttons readonly.
</ParamField>

<ParamField path="color" type="string">
  Color of the selected radio button.
</ParamField>

<ParamField path="trueIcon" type="string" default="$radioOn">
  Icon to display when a radio is selected.
</ParamField>

<ParamField path="falseIcon" type="string" default="$radioOff">
  Icon to display when a radio is not selected.
</ParamField>

<ParamField path="type" type="string" default="radio">
  HTML input type attribute.
</ParamField>

## VRadio Props

<ParamField path="label" type="string">
  Label text for the radio button.
</ParamField>

<ParamField path="value" type="any">
  The value that will be set when this radio is selected.
</ParamField>

<ParamField path="color" type="string">
  Color of the radio button when selected.
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Disable this radio button.
</ParamField>

<ParamField path="trueIcon" type="string" default="$radioOn">
  Icon when selected.
</ParamField>

<ParamField path="falseIcon" type="string" default="$radioOff">
  Icon when not selected.
</ParamField>

## Disabled State

```vue
<template>
  <div>
    <!-- Entire group disabled -->
    <VRadioGroup v-model="selected" disabled>
      <VRadio label="Option 1" value="1" />
      <VRadio label="Option 2" value="2" />
    </VRadioGroup>
    
    <!-- Individual radio disabled -->
    <VRadioGroup v-model="selected2">
      <VRadio label="Enabled" value="1" />
      <VRadio label="Disabled" value="2" disabled />
    </VRadioGroup>
  </div>
</template>

<script setup>
import { ref } from 'vue'

const selected = ref('1')
const selected2 = ref('1')
</script>
```

## Dynamic Radios

```vue
<template>
  <VRadioGroup v-model="selected" label="Select a plan">
    <VRadio
      v-for="plan in plans"
      :key="plan.id"
      :label="plan.name"
      :value="plan.id"
    />
  </VRadioGroup>
</template>

<script setup>
import { ref } from 'vue'

const selected = ref(1)
const plans = [
  { id: 1, name: 'Free' },
  { id: 2, name: 'Pro' },
  { id: 3, name: 'Enterprise' },
]
</script>
```
