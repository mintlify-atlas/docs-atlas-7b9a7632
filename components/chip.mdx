# Chip

The `VChip` component is used to convey small pieces of information. Using the `closable` prop, chips can be interactive and used as filters or selections.

## Usage

```vue
<template>
  <v-chip>Default Chip</v-chip>
</template>
```

## Props

<ParamField path="text" type="string | number | boolean">
  Text content displayed in the chip
</ParamField>

<ParamField path="label" type="boolean" default="false">
  Removes the chip's border radius
</ParamField>

<ParamField path="pill" type="boolean" default="false">
  Applies pill styling (maximum border radius)
</ParamField>

<ParamField path="closable" type="boolean" default="false">
  Adds a close icon which emits the `click:close` event when clicked
</ParamField>

<ParamField path="closeIcon" type="IconValue" default="$delete">
  Icon to use for the close button
</ParamField>

<ParamField path="closeLabel" type="string" default="$vuetify.close">
  Accessibility label for the close button
</ParamField>

<ParamField path="filter" type="boolean" default="false">
  Shows a check icon when selected (requires chip group)
</ParamField>

<ParamField path="filterIcon" type="IconValue" default="$complete">
  Icon to display when filter is active
</ParamField>

<ParamField path="prependAvatar" type="string">
  Avatar image URL displayed before the text
</ParamField>

<ParamField path="prependIcon" type="IconValue">
  Icon displayed before the text
</ParamField>

<ParamField path="appendAvatar" type="string">
  Avatar image URL displayed after the text
</ParamField>

<ParamField path="appendIcon" type="IconValue">
  Icon displayed after the text
</ParamField>

<ParamField path="draggable" type="boolean" default="false">
  Makes the chip draggable
</ParamField>

<ParamField path="link" type="boolean">
  Makes the chip a link (automatically detected when using `to` or `href`)
</ParamField>

<ParamField path="to" type="string | object">
  Vue Router location for the chip link
</ParamField>

<ParamField path="href" type="string">
  Standard anchor href attribute
</ParamField>

<ParamField path="ripple" type="boolean | object" default="true">
  Applies the ripple directive when clickable
</ParamField>

<ParamField path="modelValue" type="boolean" default="true">
  Controls the visibility of the chip
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Removes the chip's interactivity
</ParamField>

<ParamField path="activeClass" type="string">
  CSS class applied when the chip is active
</ParamField>

<ParamField path="color" type="string">
  Applies specified color to the chip
</ParamField>

<ParamField path="baseColor" type="string">
  Base color applied when chip is not selected
</ParamField>

<ParamField path="density" type="string">
  Adjusts the vertical height. Options: `default`, `comfortable`, `compact`
</ParamField>

<ParamField path="elevation" type="string | number">
  Elevation level (0-24)
</ParamField>

<ParamField path="size" type="string | number">
  Sets the height and width. Options: `x-small`, `small`, `default`, `large`, `x-large`
</ParamField>

<ParamField path="rounded" type="string | number | boolean">
  Border radius of the chip
</ParamField>

<ParamField path="tag" type="string" default="span">
  Specify a custom HTML tag to use on the root element
</ParamField>

<ParamField path="variant" type="string" default="tonal">
  Applies a distinct style. Options: `flat`, `text`, `elevated`, `tonal`, `outlined`, `plain`
</ParamField>

## Events

<ParamField path="click" type="(e: MouseEvent | KeyboardEvent) => void">
  Emitted when the chip is clicked
</ParamField>

<ParamField path="click:close" type="(e: MouseEvent) => void">
  Emitted when the close button is clicked
</ParamField>

<ParamField path="update:modelValue" type="(value: boolean) => void">
  Emitted when the chip visibility changes
</ParamField>

<ParamField path="group:selected" type="(val: { value: boolean }) => void">
  Emitted when the chip selection state changes in a group
</ParamField>

## Slots

<ParamField path="default">
  The default slot for chip content. Receives selection state props when in a group
</ParamField>

<ParamField path="prepend">
  Slot for content before the text
</ParamField>

<ParamField path="append">
  Slot for content after the text
</ParamField>

<ParamField path="close">
  Slot for custom close button content
</ParamField>

<ParamField path="filter">
  Slot for custom filter icon
</ParamField>


### Closable Chips

```vue
<template>
  <div class="d-flex gap-2">
    <v-chip 
      v-for="tag in tags"
      :key="tag"
      closable
      @click:close="removeTag(tag)"
    >
      {{ tag }}
    </v-chip>
  </div>
</template>

<script setup>
import { ref } from 'vue'

const tags = ref(['Vue', 'Vuetify', 'JavaScript'])

function removeTag(tag) {
  tags.value = tags.value.filter(t => t !== tag)
}
</script>
```
