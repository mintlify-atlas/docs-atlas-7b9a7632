# VAutocomplete

The `VAutocomplete` component provides a searchable dropdown that filters items as the user types. It extends VTextField with autocomplete functionality, supporting both single and multiple selection modes.

## Basic Usage

```vue
<template>
  <VAutocomplete
    v-model="selected"
    :items="items"
    label="Select an item"
  />
</template>

<script setup>
import { ref } from 'vue'

const selected = ref(null)
const items = ['California', 'Colorado', 'Florida', 'Georgia', 'Texas']
</script>
```

## Props

<ParamField path="autoSelectFirst" type="boolean | 'exact'">
  Automatically select the first item when the menu opens. If set to 'exact', only selects when the search value exactly matches the item title.
</ParamField>

<ParamField path="clearOnSelect" type="boolean" default="false">
  Clear the search input after selecting an item (useful with multiple mode).
</ParamField>

<ParamField path="search" type="string">
  The current search query. Can be bound with `v-model:search`.
</ParamField>

<ParamField path="items" type="readonly any[]">
  Array of items to display in the dropdown.
</ParamField>

<ParamField path="itemTitle" type="string | function">
  Property name or function to determine the display text for each item.
</ParamField>

<ParamField path="itemValue" type="string | function">
  Property name or function to determine the value for each item.
</ParamField>

<ParamField path="returnObject" type="boolean" default="false">
  Return the entire item object as the value instead of just the item value.
</ParamField>

<ParamField path="multiple" type="boolean" default="false">
  Allow multiple item selection.
</ParamField>

<ParamField path="chips" type="boolean" default="false">
  Display selected items as chips.
</ParamField>

<ParamField path="closableChips" type="boolean" default="false">
  Allow chips to be closed/removed.
</ParamField>

<ParamField path="hideSelected" type="boolean" default="false">
  Hide selected items from the list.
</ParamField>

<ParamField path="hideNoData" type="boolean" default="false">
  Hide the menu when there are no items to display.
</ParamField>

<ParamField path="menuIcon" type="string" default="$dropdown">
  Icon to display for the dropdown menu indicator.
</ParamField>

<ParamField path="noDataText" type="string" default="$vuetify.noDataText">
  Text to display when no items match the search.
</ParamField>

<ParamField path="filterKeys" type="string[]" default="['title']">
  Item properties to use when filtering.
</ParamField>

<ParamField path="openOnClear" type="boolean" default="false">
  Open the menu when the clear icon is clicked.
</ParamField>

## Multiple Selection

```vue
<template>
  <VAutocomplete
    v-model="selected"
    :items="items"
    label="Select multiple"
    multiple
    chips
    closable-chips
  />
</template>

<script setup>
import { ref } from 'vue'

const selected = ref([])
const items = ['Item 1', 'Item 2', 'Item 3', 'Item 4', 'Item 5']
</script>
```

## Custom Filtering

```vue
<template>
  <VAutocomplete
    v-model="selected"
    :items="items"
    item-title="name"
    item-value="id"
    label="Search users"
    :filter-keys="['name', 'email']"
  />
</template>

<script setup>
import { ref } from 'vue'

const selected = ref(null)
const items = [
  { id: 1, name: 'John Doe', email: 'john@example.com' },
  { id: 2, name: 'Jane Smith', email: 'jane@example.com' },
  { id: 3, name: 'Bob Johnson', email: 'bob@example.com' },
]
</script>
```

## Auto Select First

```vue
<template>
  <VAutocomplete
    v-model="selected"
    :items="items"
    label="Auto select first"
    auto-select-first
  />
</template>
```

## Events

<ParamField path="update:modelValue" type="(value: any) => void">
  Emitted when the selected value changes.
</ParamField>

<ParamField path="update:search" type="(value: string) => void">
  Emitted when the search query changes.
</ParamField>

<ParamField path="update:menu" type="(value: boolean) => void">
  Emitted when the menu opens or closes.
</ParamField>

<ParamField path="update:focused" type="(focused: boolean) => void">
  Emitted when the input focus state changes.
</ParamField>

## Slots

<ParamField path="item" type="{ item, index, props }">
  Customize the appearance of list items.
</ParamField>

<ParamField path="chip" type="{ item, index, props }">
  Customize the appearance of selected chips.
</ParamField>

<ParamField path="selection" type="{ item, index }">
  Customize how selected items are displayed.
</ParamField>

<ParamField path="no-data" type="{}">
  Content to display when no items are available.
</ParamField>

<ParamField path="prepend-item" type="{}">
  Content to display before the list items.
</ParamField>

<ParamField path="append-item" type="{}">
  Content to display after the list items.
</ParamField>
